LIST(APPEND COMPOSITE_OBJECTS_CC_SOURCES
   main.c
   simple.c
   comp.c
   )

   ${EINA_INCLUDE_DIRS}
   ${EVAS_INCLUDE_DIRS}
   ${CMAKE_SOURCE_DIR}/lib
   )

   ${EINA_LDFLAGS_OTHER}

   )
